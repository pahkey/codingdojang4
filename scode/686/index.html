package main

import "fmt"

func Ordinal(number int) (result string) {
    // *1*이 아닐 때
    if (number / 10) % 10 != 1 {
        switch number % 10 {
        case 1:
            result = "st"
        case 2:
            result = "nd"
        case 3:
            result = "rd"
        default:
            result = "th"
        }
    } else {
        // *1*일 때
        result = "th"
    }
    return
}

func main() {
    for {
        fmt.Print("숫자를 입력하세요 (0 입력 = 종료): ")
        var number int

        _, _ = fmt.Scanf("%d", &number)

        if number == 0 {
            break
        }
        fmt.Printf("%d%s\n", number, Ordinal(number))
    }
}
