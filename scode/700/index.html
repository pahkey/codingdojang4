### 자동 게임
from random import randrange
def myCall(comCall):
    for i in range(1, 4):
        myNum = comCall + i
        if myNum % 4 == 3:
            print(f"You: {myNum}")
            return myNum
    myNum = comCall + randrange(1, 4)
    print(f"You: {myNum}")
    return myNum

comCall = randrange(1, 4) ### 컴이 3부터 계속 지는 수를 말하면 실패
print(f"Com: {comCall}")
while comCall < 30:
    myNum = myCall(comCall)
    if myNum == 31:
        print('Human lost')
        break
    randNum = randrange(1, 4)
    comCall = myNum + randNum
    print(f"Com: {comCall}")
