def longest_common_str
  a,b = (1..2).map {gets.chop}.sort_by(&:size)
  cmn = ->s { (s...a.size).reduce("") {|_,e| b.include?(a[s..e])? a[s..e] : break } }
  lcs = (0...a.size).reduce("") {|gcs,s| [gcs, cmn[s]||""].max_by(&:size) }
  puts lcs.size, lcs
end
