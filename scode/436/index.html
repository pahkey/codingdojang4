def ugly(n):
    re = 1
    n2, n3, n5 = [], [], []
    for i in range(n-1):
        n2 += [re*2]
        n3 += [re*3]
        n5 += [re*5]
        re = min(n2[0], n3[0], n5[0])
        if re == n2[0]:
            del n2[0]
        if re == n3[0]:
            del n3[0]
        if re == n5[0]:
            del n5[0]
    return re

# a HeapSet is a heap containing unique items
class HeapSet:
    def __init__(self, *args):
        self.heap = []
        self.set = set()
        self.push(*args)

    def push(self, *args):
        for item in args:
            if item not in self.set:
                heapq.heappush(self.heap, item)
                self.set.add(item)

    def pop(self):
        item = heapq.heappop(self.heap)
        self.set.remove(item)
