#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define MIN 1   // 정답의 최소값
#define MAX 100 // 정답의 최대값

int main(void)
{
    int user_num; // 유저가 입력한 값
    int cnt = 1; // 숫자를 입력한 횟수
    int crt;    // 정답


    srand(time(NULL));
    crt = (rand() % MAX) + MIN;  // MIN~MAX중 랜덤한 숫자 리턴

    while(1) {  
        printf("%d~%d 숫자 입력:", MIN, MAX);
        scanf("%d", &user_num);

        if(crt == user_num) {
            printf("정답입니다! %d회 만에 맞췄어요.\n", cnt);
            break;
        }

        else if(crt < user_num) {
            puts("DOWN");   
        }

        else {
            puts("UP");
        }
        cnt++;
    }

    return 0;
}
