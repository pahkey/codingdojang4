  x <- as.character(x)

  number <- nchar(x)
  if(number < 4) return(x)
  first <- ifelse(number%%3 == 0, number%%3, number%%3 + 1)

  tmpvec[-seq(first, number, 4)] <- x %>% str_split('') %>% unlist
