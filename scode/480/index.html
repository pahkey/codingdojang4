def do(n):
  buffer = [' '] * (n*n)
  buffer[0::n] = ['N'] * n    ## 왼쪽 세로선
  buffer[n-1::n] = ['N'] * n  ## 오른쪽 세로선
  buffer[n+1::n+1] = ['N'] * (n-1)  ## 대각선
  return '\n'.join(''.join(buffer[i*n:i*n+n]) for i in range(n))
