def sieve(n):
    if n < 2:
        return []
    s = [0, 0] + [1] * (n - 1)
    for i in range(2, int(n**.5)+1):
        if s[i]:
            s[i*2::i] = [0] * ((n - i)//i)
    return [i for i, v in enumerate(s) if v]

    // 소수가 아닌 값들은 1로 셋팅

    clock_t start, end;

    if n < 2:
        return []

    range_ = set(range(2, n + 1))

        base = min(range_)
        if base ** 2 > n:

        range_ -= set(range(base, n + 1, base))
