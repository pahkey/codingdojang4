import unittest

def IsDuplicated(number):
    # 중복된 원소를 식별하기 위해 집합을 사용한다.
    num2set = set(number)

    # 입력된 숫자길이가 10보다 작으면 빠진 숫자가 있는 것이고,
    # 또는 10이상이면 최소한 한자리는 중복될수 밖에 없다.
    if not len(number) == 10:
        return False

    # 집합에서는 중복된 원소가 존재하지 않으므로,
    # 중복을 피ㅏ려면 반드시 갯수가 10이어야 한다.
    if not len(num2set) == 10:
        return False

    return True

class DuplicateNumberTest(unittest.TestCase):

        self.assertEqual(IsDuplicated('0123456789'),True)
        self.assertEqual(IsDuplicated('01234'), False)
        self.assertEqual(IsDuplicated('01234567890'), False)
        self.assertEqual(IsDuplicated('6789012345'), True)
        self.assertEqual(IsDuplicated('012322456789'), False)
