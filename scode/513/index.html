        {

                count += Fibonacci(i);

            }

        public int Fibonacci(int n)
        {
            var first = 0;
            var second = 1;
            for (int i = 1; i <= n; i++)
            {
                var temp = first;
                first = second;
                second += temp;
            }
            return second;
        }

void base(int b, int v, int* a, int n){
    for(int i = 0; i < n; i++){

        return;

}

int solve(int n,int j){
    // n: 계단 수, j: 한번에 올라갈 수 있는 최대 계단 수 
    if (n == 0) return 1;

}

var climbing = function (n, j) {
    cache = [0, 1];
    for (let i = 2; i <= j; i++) {

def stair(n, j):
    result = 0
    if n == 0:
        result += 1
    elif n < j:
        for i in range(n):

def f(n,j):
  if n == 0: return 1
  if n < 0: return 0
  _sum = 0
  for i in range(n-j, n): _sum += f(i, j)
  return _sum

def rec_f(n0, j0):
    case = 0
    if n0 == 0 or n0 == 1:
        return 1
    else:
        for i in range(min(n0, j0)):
            case += rec_f(n0 - i - 1, j0)
    return case

def stairway(N, J) :
    queue, res = [[i+1] for i in range(J)], []

    while len(queue) != 0 :
        fst = queue.pop(0)
        if sum(fst) == N : 
            res.append(fst) ## 계단을 모두 올랐을 경우 : print
            continue
        for st in range(1, J+1) :
            if sum(fst)+st <= N :
                queue.append(fst+[st])
    return len(res)
print(stairway(5, 3))
