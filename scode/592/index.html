            // Shuffle Generic List

            List<T> newList = new List<T>();
            while (sourceList.Count > 0)
            {
                int idx = rnd.Next(0, sourceList.Count);
                newList.Add(sourceList[idx]); sourceList.RemoveAt(idx);
            }
            // flip fingers
            int len = newList.Count;

        }
