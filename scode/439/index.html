Naive Solution

이 문제를 풀기 위한 가장 간단한 풀이는 각 위치에 대해서 슬라이딩 윈도를 직접 대어 보고 최대값과 최소값이 무엇인지 찾는 것입니다. 2중 for문을 통해서 구현하면 되겠죠. O(NK)입니다.

Indexed Tree 인덱스 트리라는 자료구조는 완전 이진 트리의 일종인데, 알고리즘 경시에 참여하는 사람들 사이에서 통용되는 이름입니다. 구간 트리라고도 불리는데 일반적인 구간 트리와는 약간 다릅니다. 일단 이 트리는 완전 이진 트리이므로 N보다 크거나 같은 가장 작은 2의 지수를 M이라 할 때, 2*M개의 저장공간을 필요로 합니다.

각 노드는 자신이 가리키는 구간을 ‘대표하는’ 값을 지니게 됩니다. 예를 들어 이 문제의 경우 트리의 루트 노드는 Min( Array[0..M-1] ) 또는 대응되는 Max의 값을 가지게 되겠죠. 루트의 왼쪽 노드는 0 ~ M/2-1까지의 구간을 대표하고, 루트의 오른쪽 노드는 M/2 ~ M-1까지를 대표하고.. 이런 식으로 깊이가 한단계 깊어질 때마다 노드가 대표하는 구간의 크기가 절반씩 줄어듭니다. 깊이가 log M이 되면 각각 길이 1의 구간을 대표하고 있게 되죠. (그 위치에 해당하는 Array의 값을 지닌다는 뜻)

트리의 갱신은 일단 가장 끝부분의 값을 바꾸고, 부모 노드를 하나씩 타고 올라오면서 값을 갱신해 줍니다. 두 자식의 값을 보고 자기자신을 바꾸는 recursive 함수를 작성하면 되겠죠?

이렇게 하면 임의의 구간에 대한 값을 구하는 건 lg M 개의 노드만 선택하면 어떤 구간이라도 커버가 가능하기 때문에… O(lg M) = O(lg N)입니다. M < 2*N이니까요.

#include<vector>
#include<algorithm>
#include<functional>
#include<iostream>
using namespace std;

class IndexedTree{
private:
    vector<int> minTree;
    vector<int> maxTree;

    int treeBegin;  
    void initTree(){
        int here;
        for(here = treeBegin - 1; here >= 1; --here){
            const int leftChild = here * 2;
            const int rightChild = leftChild + 1;

            minTree[here] = minTree[leftChild] < minTree[rightChild]? minTree[leftChild] : minTree[rightChild];
            maxTree[here] = maxTree[leftChild] > maxTree[rightChild]? maxTree[leftChild] : maxTree[rightChild];
        }       
    }
public:
    IndexedTree(vector<int> data){
        for(treeBegin = 1; treeBegin < data.size(); treeBegin <<=1);

        minTree.resize(treeBegin * 2);
        maxTree.resize(treeBegin * 2);

        copy(data.begin(), data.end(), minTree.begin() + treeBegin);
        copy(data.begin(), data.end(), maxTree.begin() + treeBegin);

        initTree();
    }

    int getMin(int firstIndex, int lastIndex){
        firstIndex += treeBegin;
        lastIndex += treeBegin;

        int result = 0x7fffffff;
        while(firstIndex < lastIndex){
            result = min(result, minTree[firstIndex]);
            result = min(result, maxTree[lastIndex]);

            firstIndex = (firstIndex + 1) >> 1;
            lastIndex = (lastIndex - 1) >> 1;
        }
        if(firstIndex == lastIndex)
            return min(result, minTree[firstIndex]);
        return result;
    }

    int getMax(int firstIndex, int lastIndex){
        firstIndex += treeBegin;
        lastIndex += treeBegin;

        int result = -2147483648;
        while(firstIndex < lastIndex){
            result = max(result, maxTree[firstIndex]);
            result = max(result, maxTree[lastIndex]);

            firstIndex = (firstIndex + 1) >> 1;
            lastIndex = (lastIndex - 1) >> 1;
        }
        if(firstIndex == lastIndex)
            return max(result, maxTree[firstIndex]);
        return result;
    }
};

int main(){
    int n, k, i;
    cin >> n >> k;

    vector<int> data(n);
    for(i = 0; i < n; ++i){
        cin >> data[i];
    }

    vector<int> mins;
    vector<int> maxs;
    IndexedTree it(data);

    mins.reserve(n-k+1);
    mins.reserve(n-k+1);
    for(i = k-1; i < n; ++i){
        mins.push_back( it.getMin(i-k+1, i) );
        maxs.push_back( it.getMax(i-k+1, i) );
    }

    for(i = 0; i < mins.size(); ++i){
        cout << mins[i] << " ";
    }
    cout << "\n";
    for(i = 0; i < maxs.size(); ++i){
        cout << maxs[i] << " ";
    }
    return 0;
}

Min[0..3] Min[1..3] Min[2..3] A[3] Min[3..4] Min[3..5] Min[3..6]

그런데 이걸 잘 보시면요. 이 배열만으로 슬라이딩 윈도우 내에서 최소값이 무엇인지 다 알 수 있어요.

Min[0..3]은 배열 안에 있고요. Min[1..4] = Min[1..3]과 Min[3..4] 중 작은 값이죠. Min[2..5]는 Min[2..3]과 Min[3..5]중에 작은 값이죠…

어헣 간단하죠? 2*K-1개씩 배열을 나누면 대충 O(N/K)개의 부분배열이 나오죠? 각각의 부분배열에서 최대값과 최소값을 구하는 것은 O(K)만에 되니까 O(N/K) * O(K) = O(N) 이 되는 것입니다.

#include<cstdio>
#include<vector>
using namespace std;

int N,K;
inline int min(int a,int b){return a>b?b:a;}
inline int max(int a,int b){return a>b?a:b;}

const int maxn=1000000;
int data[2*maxn],deque_min[2*maxn],deque_max[2*maxn],M[maxn];

int main(){
    //var;
    int pivot,i,j,c;

    //code
    scanf("%d %d",&N,&K);   

    for(i=0;i<N;i++)
        scanf("%d",&data[i]);

    for(c=0,pivot=K-1;pivot<N;pivot+=K){
        deque_max[K-1]=deque_min[K-1]=data[pivot];

        for(i=1;i<K;i++){
            deque_min[K-i-1]=min(deque_min[K-i],data[pivot-i]);
            deque_max[K-i-1]=max(deque_max[K-i],data[pivot-i]);

            if(pivot+i-1 >= N-1) {
                deque_max[K+i-1]=deque_min[K+i-1]=0x7fffffff;
            }
            else {
                deque_min[K+i-1]=min(deque_min[K+i-2],data[pivot+i]);
                deque_max[K+i-1]=max(deque_max[K+i-2],data[pivot+i]);
            }
        }
        for(i=0,j=K-1;i<K;++i,++j){
            if(++c > N-K+1) break;  
            M[c-1]=max(deque_max[i],deque_max[j]);
            printf("%d ",min(deque_min[i],deque_min[j]));
        }
    }
    printf("\n");
    for(i=0;i<N-K+1;++i)
        printf("%d ",M[i]);

    return 0;
}

import bisect

def slide(src, k):
    window = map(int, src.split())
    last_index = len(window)-k
    r1, r2 = [], []

    current_window = window[0:k]
    current_window.sort()
    r1.append(current_window[0])
    r2.append(current_window[-1])

    for i in range(last_index):
        current_window.remove(window[i])
        bisect.insort(current_window, window[k+i])
        r1.append(current_window[0])
        r2.append(current_window[-1])

    return r1,r2

        static void DisplaySlidingMinMax(int windowLength, List<int> inputArray)
        {
            Queue<int> window = new Queue<int>(inputArray.Take(windowLength - 1));
            List<int> maxValues = new List<int>();
            List<int> minValues = new List<int>();
            // sliding window 큐에 새로운 입력 배열의 다음 항목 enqueue, 기존 항목 dequeue
            // 최대, 최소값 확인
            for (int idx = windowLength - 1; idx < inputArray.Count; idx++)
            {
                window.Enqueue(inputArray[idx]);
                if (window.Count > windowLength) { _ = window.Dequeue(); }
                maxValues.Add(window.Max());
                minValues.Add(window.Min());
            }

            Console.WriteLine(string.Join(" ", minValues));
            Console.WriteLine(string.Join(" ", maxValues));
        }
