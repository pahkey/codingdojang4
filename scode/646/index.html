'use strict';

function determineCharsAndNumbers(input) {
    let numbers = [];
    let chars = [];

    Array.from(input).forEach(char => {
        if (char * 1 === parseInt(char)) {
            numbers.push(char);
        } else {
            chars.push(char);
        }
    });

    return { numbers, chars };
};

const {numbers, chars} = determineCharsAndNumbers('c910m6ia 1ho');
