def isMultipleOfN(x, n):
    l = len(str(x))
    y = n * (10**(l-1))
    while x >= n:
        if y > x:
            x, y = x, y-x
        else:
            x, y = y, x-y
    if x == 0:
        return True
    else:
        return False


print(isMultipleOfN(432, 2))
print(isMultipleOfN(4, 3))
print(isMultipleOfN(635, 5))
print(isMultipleOfN(421, 7))
